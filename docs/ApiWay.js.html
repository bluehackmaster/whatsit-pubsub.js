import User from './User';

/**
 * WiPubSub encapsulates the functionality to create various API wrapper objects.
 */
class WiPubSub {
   /**
    * Create a new WiPubSub.
    * @param {Requestable.auth} [auth] - the credentials to authenticate to WiPubSub. If auth is
    *                                  not provided requests will be made unauthenticated
    * @param {string} [apiBase=https://api.WiPubSub.io] - the base WiPubSub API URL
    */
   // constructor(auth, apiBase = 'https://api.WiPubSub.io') {
    constructor(auth, apiBase = 'http://127.0.0.1:3000') {
      this.__apiBase = apiBase;
      this.__auth = auth || {};
   }

   /**
    * Create a new User wrapper
    * @param {string} [user] - the name of the user to get information about
    *                        leave undefined for the authenticated user
    * @return {User}
    */
   getUser(user) {
      return new User(user, this.__auth, this.__apiBase);
   }
}

module.exports = WiPubSub;
